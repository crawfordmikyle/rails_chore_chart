Schema Layout

tables
  User
    attributes
      email
      password_digest
      name

    associations
      has_many UserTeams
      has_many Teams through UserTeam
      has_many Friendships
      has_many Friends through Friendships
      has_many Tasks
      has_many Chores through Tasks

  Team
    attributes
      name

    associations
      has_many UserTeams
      has_many Users through UserTeam
      has_many Chores


  UserTeam
    attributes
      user_id
      team_id
      accepted?
    associations
      belongs_to User
      belongs_to Team


  Chore
    attributes
      name
      value
      due_date
      status
    associations
      belongs_to Team

  Tasks
    attributes
      user_id
      chore_id

    associations
      belongs_to User
      belongs_to Chore

  Friendships
    attributes
      user_id
      friend_id
      accepted?
    associations
      belongs_to User
      belongs_to Friend class User
